1. Get the application URL by running these commands:
{{- if .Values.ingress.enabled }}
  https://{{ index .Values.ingress.hosts 0 "host" }}
{{- else if contains "NodePort" .Values.service.type }}
  export NODE_PORT=$(kubectl get --namespace {{ .Release.Namespace }} -o jsonpath="{.spec.ports[0].nodePort}" services {{ include "spring-ai-faq-retrieval.fullname" . }})
  export NODE_IP=$(kubectl get nodes --namespace {{ .Release.Namespace }} -o jsonpath="{.items[0].status.addresses[0].address}")
  echo http://$NODE_IP:$NODE_PORT
{{- else if contains "LoadBalancer" .Values.service.type }}
  NOTE: It may take a few minutes for the LoadBalancer IP to be available.
  You can watch the status of by running 'kubectl get --namespace {{ .Release.Namespace }} svc -w {{ include "spring-ai-faq-retrieval.fullname" . }}'
  export SERVICE_IP=$(kubectl get svc --namespace {{ .Release.Namespace }} {{ include "spring-ai-faq-retrieval.fullname" . }} --template "{{"{{ range (index .status.loadBalancer.ingress 0) }}{{.}}{{ end }}"}}")
  echo http://$SERVICE_IP:{{ .Values.service.port }}
{{- else if contains "ClusterIP" .Values.service.type }}
  export POD_NAME=$(kubectl get pods --namespace {{ .Release.Namespace }} -l "app.kubernetes.io/name={{ include "spring-ai-faq-retrieval.name" . }},app.kubernetes.io/instance={{ .Release.Name }}" -o jsonpath="{.items[0].metadata.name}")
  export CONTAINER_PORT=$(kubectl get pod --namespace {{ .Release.Namespace }} $POD_NAME -o jsonpath="{.spec.containers[0].ports[0].containerPort}")
  echo "Visit http://127.0.0.1:8080 to use your application"
  kubectl --namespace {{ .Release.Namespace }} port-forward $POD_NAME 8080:$CONTAINER_PORT
{{- end }}

2. Check deployment status:
  kubectl --namespace {{ .Release.Namespace }} rollout status deployment/{{ include "spring-ai-faq-retrieval.fullname" . }}

3. Get service information:
  kubectl get svc --namespace {{ .Release.Namespace }} {{ include "spring-ai-faq-retrieval.fullname" . }}

4. View logs:
  kubectl --namespace {{ .Release.Namespace }} logs deployment/{{ include "spring-ai-faq-retrieval.fullname" . }} -f

5. Access API documentation:
  curl http://{{ index .Values.ingress.hosts 0 "host" }}/swagger-ui.html

6. Useful debugging commands:
  - Check pod status: kubectl get pods -n {{ .Release.Namespace }}
  - Describe deployment: kubectl describe deployment {{ include "spring-ai-faq-retrieval.fullname" . }} -n {{ .Release.Namespace }}
  - Check events: kubectl get events -n {{ .Release.Namespace }} --sort-by='.lastTimestamp'
  - Check resource metrics: kubectl top pods -n {{ .Release.Namespace }}
