1. Get the application URL by running these commands:
{{- if .Values.ingress.enabled }}
  https://{{ index .Values.ingress.hosts 0 "host" }}
{{- else if contains "NodePort" .Values.serviceConfig.type }}
  export NODE_PORT=$(kubectl get --namespace {{ .Release.Namespace }} -o jsonpath="{.spec.ports[0].nodePort}" services {{ include "langchain-rag-services.fullname" . }})
  export NODE_IP=$(kubectl get nodes --namespace {{ .Release.Namespace }} -o jsonpath="{.items[0].status.addresses[0].address}")
  echo http://$NODE_IP:$NODE_PORT
{{- else if contains "LoadBalancer" .Values.serviceConfig.type }}
  NOTE: It may take a few minutes for the LoadBalancer IP to be available.
  You can watch the status of by running 'kubectl get --namespace {{ .Release.Namespace }} svc -w {{ include "langchain-rag-services.fullname" . }}'
  export SERVICE_IP=$(kubectl get svc --namespace {{ .Release.Namespace }} {{ include "langchain-rag-services.fullname" . }} --template "{{"{{ range (index .status.loadBalancer.ingress 0) }}{{.}}{{ end }}"}}")
  echo http://$SERVICE_IP:{{ .Values.serviceConfig.port }}
{{- else if contains "ClusterIP" .Values.serviceConfig.type }}
  export POD_NAME=$(kubectl get pods --namespace {{ .Release.Namespace }} -l "app.kubernetes.io/name={{ include "langchain-rag-services.name" . }},app.kubernetes.io/instance={{ .Release.Name }}" -o jsonpath="{.items[0].metadata.name}")
  export CONTAINER_PORT=$(kubectl get pod --namespace {{ .Release.Namespace }} $POD_NAME -o jsonpath="{.spec.containers[0].ports[0].containerPort}")
  echo "Visit http://127.0.0.1:8080 to use your application"
  kubectl --namespace {{ .Release.Namespace }} port-forward $POD_NAME 8080:$CONTAINER_PORT
{{- end }}

2. Check deployment status:
  kubectl rollout status deployment/{{ include "langchain-rag-services.fullname" . }} -n {{ .Release.Namespace }}

3. View logs:
  kubectl logs -n {{ .Release.Namespace }} deployment/{{ include "langchain-rag-services.fullname" . }} -f

4. Access metrics:
  kubectl port-forward -n {{ .Release.Namespace }} deployment/{{ include "langchain-rag-services.fullname" . }} 9090:9090
  Visit http://localhost:9090/metrics for Prometheus metrics

5. Access API documentation:
  kubectl port-forward -n {{ .Release.Namespace }} deployment/{{ include "langchain-rag-services.fullname" . }} 8000:8000
  Visit http://localhost:8000/api/docs for interactive API docs

6. Scale the deployment:
  kubectl scale deployment {{ include "langchain-rag-services.fullname" . }} -n {{ .Release.Namespace }} --replicas=<NUM>

7. Watch pod events:
  kubectl get pods -n {{ .Release.Namespace }} -l "app.kubernetes.io/name={{ include "langchain-rag-services.name" . }}" -w

For more help, visit: https://helm.sh/docs
